function formValidate() { //var err = true; var name = document.getElementById("name").value; //Alphabets and space(' ') no special characters min 3 and max 20 characters. var namePattern = /^[A-Za-z ]{3,20}$/; if(!namePattern.test(name)) { $('#nameErrorMessage').addClass('visible') $('#name').addClass('error') //return false; }else { $('#nameErrorMessage').removeClass('visible'); $('#name').removeClass('error'); } var email = document.getElementsByName("email")[0].value; //email pattern var emailPattern = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/; if(!emailPattern.test(email)) { $('#emailErrorMessage').addClass('visible'); $('#email').addClass('error'); //return false; }else { $('#emailErrorMessage').removeClass('visible'); $('#email').removeClass('error'); } var username = document.getElementsByName("username")[0].value; //Alphabets and space(' ') no special characters min 3 and max 20 characters. var usernamePattern = /^[A-Za-z ]{3,20}$/; if(!usernamePattern.test(username)) { $('#usernameErrorMessage').addClass('visible'); $('#username').addClass('error'); //return false; }else { $('#usernameErrorMessage').removeClass('visible'); $('#username').removeClass('error'); } var password = document.getElementById("password").value; var passwordPattern = /^[A-Za-z ]{3,20}$/; if(!passwordPattern.test(password)) { $('#passwordErrorMessage').addClass('visible'); $('#password').addClass('error'); return false; }else { $('#passwordErrorMessage').removeClass('visible') $('#password').removeClass('error'); } alert("You entered a valid data!"); return true; }
